# Creating dictionaries student = { "name": "Alice Johnson", "age": 20, "gpa": 3.8, "courses": ["COMP1005", "MATH1010"] } # Accessing values name = student["name"] # "Alice Johnson" age = student.get("age") # 20 (safe access) major = student.get("major", "Undeclared") # Default value # Modifying dictionaries student["gpa"] = 3.9 # Update existing key student["major"] = "Computer Science" # Add new key del student["age"] # Remove key

Merge dictionaries
# Dictionary methods and iteration book_ratings = { "Python Programming": 4.5, "Data Structures": 4.2, "Algorithms": 4.8 } # Iterating through dictionaries for title in book_ratings.keys(): print(f"Book: {title}") for rating in book_ratings.values(): print(f"Rating: {rating}") for title, rating in book_ratings.items(): print(f"{title}: {rating}") # Dictionary comprehensions squared = {x: x**2 for x in range(5)} # {0: 0, 1: 1, 2: 4, 3: 9, 4: 16} # Nested dictionaries course_info = { "COMP1005": { "instructor": "Dr. Smith", "students": 25, "schedule": "Mon-Wed 9:00 AM" } }

Dictionaries are like labeled containers - you can quickly find items using their labels (keys) instead of positions!
