# Class definition class Student: """Student class for university management""" def __init__(self, name, student_id, gpa): self.name = name self.student_id = student_id self.gpa = gpa self.courses = [] def __str__(self): return f"{self.name} (ID: {self.student_id}, GPA: {self.gpa})" def add_course(self, course): self.courses.append(course) def get_classification(self): if self.gpa >= 3.7: return "First Class" elif self.gpa >= 3.0: return "Upper Second" else: return "Lower Second"
