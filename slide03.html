# Function definition def calculate_gpa(grades): """Calculate GPA from letter grades""" total_points = 0 for grade in grades: if grade == 'A': total_points += 4.0 elif grade == 'B': total_points += 3.0 gpa = total_points / len(grades) return gpa # Function call student_grades = ['A', 'B', 'A', 'A'] result = calculate_gpa(student_grades) print(f"GPA: {result}")
